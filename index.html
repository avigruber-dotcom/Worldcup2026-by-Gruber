function baseRowsForGroup(group){
  const rows = state.groups[group].map((team,idx)=>({team,idx,p:0,w:0,d:0,l:0,gf:0,ga:0,gd:0,pts:0}));
  state.groupMatches.filter(m => m.group===group && complete(m)).forEach(m=>{
    const A = rows.find(r=>r.team===m.teamA), B = rows.find(r=>r.team===m.teamB);
    if(!A || !B) return;
    const ga=Number(m.goalsA), gb=Number(m.goalsB);
    A.p++; B.p++; A.gf+=ga; A.ga+=gb; B.gf+=gb; B.ga+=ga;
    if(ga>gb){A.w++; B.l++; A.pts+=3;} else if(gb>ga){B.w++; A.l++; B.pts+=3;} else {A.d++; B.d++; A.pts++; B.pts++;}
  });
  rows.forEach(r=>r.gd=r.gf-r.ga);
  return rows;
}

function resolveSlot(slot, tables, thirdPool){
  if(typeof slot === "number" || String(slot).startsWith("L")) return null;
  if(String(slot).includes(":")){
    const allowed = slot.split(":")[1].split("/");
    const cand = thirdPool.find(t => t.qualified && allowed.includes(t.group) && !t.used);
    if(cand){ cand.used = true; return `${cand.team} — מקום 3 בית ${cand.group}`; }
    return `מקום 3: ${allowed.join("/")}`;
  }
  const pos = Number(slot[0])-1, g = slot[1];
  return tables[g]?.[pos]?.team || slot;
}
function seededKnockout(tables, thirdList){
  const byId = {};
  const thirdPool = thirdList.map(t=>({...t, used:false}));
  for (const m of state.knockout) {
    let teamA, teamB;
    if(typeof m.a === "number") teamA = winnerOf(byId[m.a]);
    else if(String(m.a).startsWith("L")) teamA = loserOf(byId[Number(String(m.a).slice(1))]);
    else teamA = resolveSlot(m.a, tables, thirdPool);

    if(typeof m.b === "number") teamB = winnerOf(byId[m.b]);
    else if(String(m.b).startsWith("L")) teamB = loserOf(byId[Number(String(m.b).slice(1))]);
    else teamB = resolveSlot(m.b, tables, thirdPool);

    byId[m.id] = {...m, teamA, teamB};
  }
  return Object.values(byId);
}

function outcome(a,b){
  a = Number(a); b = Number(b);
  if(a > b) return "A";
  if(b > a) return "B";
  return "D";
}
function hasPrediction(m){
  return m && m.predA !== "" && m.predB !== "" && !isNaN(Number(m.predA)) && !isNaN(Number(m.predB));
}
function predictionScoreFor(m){
  if(!complete(m) || !hasPrediction(m)) return null;
  const exact = Number(m.predA) === Number(m.goalsA) && Number(m.predB) === Number(m.goalsB);
  const sameOutcome = outcome(m.predA, m.predB) === outcome(m.goalsA, m.goalsB);
  return { exact, sameOutcome, points: exact ? 3 : sameOutcome ? 1 : 0 };
}
function predictionSummary(koList){
  const all = [...state.groupMatches, ...koList];
  let checked = 0, exact = 0, outcomeOnly = 0, points = 0;
  all.forEach(m => {
    const s = predictionScoreFor(m);
    if(!s) return;
    checked++;
    points += s.points;
    if(s.exact) exact++;
    else if(s.sameOutcome) outcomeOnly++;
  });
  return {checked, exact, outcomeOnly, points};
}
